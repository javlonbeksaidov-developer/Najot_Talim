import random

chars = ["!@#$%^&*"]
ball = 0

random_char = chars[
    random.randint(0, 7)
]



ishora = ""


while ishora:
    belgi = input("Belgi tanlang: (!@#$%^&*) ")
    if belgi == random_char:
        ball += 10
    else:
        ishora = False
        
print(f"Sizning balingiz {ball}")
